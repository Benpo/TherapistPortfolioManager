Per D-14, add a license management link in the header area. Add a small key/license icon link to the `.header-actions` div, next to the language picker and theme toggle.

In `renderNav()` or after it in the init flow, add a license link element. The cleanest approach: add it in the `header-actions` div rendering. Since `header-actions` is static HTML in each page's HTML file, instead add it dynamically in app.js after `initThemeToggle()`:

In the `init()` function (around line 115), after `initThemeToggle()`:

```javascript
// Add license management link to header
(function initLicenseLink() {
  var actions = document.querySelector('.header-actions');
  if (!actions) return;
  var link = document.createElement('a');
  link.href = './license.html';
  link.className = 'header-license-link';
  link.setAttribute('aria-label', App.t('nav.license') || 'License');
  link.setAttribute('title', App.t('nav.license') || 'License');
  // Key SVG icon (16x16)
  link.innerHTML = '<svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M21 2l-2 2m-7.61 7.61a5.5 5.5 0 1 1-7.778 7.778 5.5 5.5 0 0 1 7.777-7.777zm0 0L15.5 7.5m0 0l3 3L22 7l-3-3m-3.5 3.5L19 4"/></svg>';
  actions.prepend(link);
})();
```
