    function checkBackupSchedule(opts) {
      opts = opts || {};
      var now = (typeof opts.now === 'number') ? opts.now : Date.now();
      var intervalMs = getScheduleIntervalMs();
      if (intervalMs === null) return; // OFF — Plan 04 banner-suppression is also OFF; let the existing 7-day banner handle it (it will, because the suppression guard early-returns only when intervalMs !== null)

      var lastExportRaw;
      try { lastExportRaw = localStorage.getItem('portfolioLastExport'); }
      catch (_) { return; }
      var lastExport = lastExportRaw ? Number(lastExportRaw) : 0;

      var dueAt = lastExport + intervalMs;
      if (now < dueAt) return;

      // 1-hour debounce so the prompt does not re-fire on every visibilitychange flip in the same hour.
      var lastPromptKey = 'portfolioBackupSchedulePromptedAt';
      var lastPrompt;
      try { lastPrompt = Number(localStorage.getItem(lastPromptKey)) || 0; }
      catch (_) { lastPrompt = 0; }
      if (now - lastPrompt < 60 * 60 * 1000) return;

      try { localStorage.setItem(lastPromptKey, String(now)); } catch (_) {}

      // Open the Backup & Restore modal (Plan 02 exposes openBackupModal on window).
      // The modal's Export section IS the interval-end prompt — clicking Export goes through the encrypt-or-skip flow.
      // D-20: no silent folder-write; user must press Export themselves.
      try {
        if (typeof window !== 'undefined' && typeof window.openBackupModal === 'function') {
          window.openBackupModal();
        }
      } catch (_) {}
    }

    function canEnableSchedule(mode) {
      if (mode === 'off') return true;
      try { return localStorage.getItem('portfolioBackupSchedulePasswordAcked') === 'true'; }
      catch (_) { return false; }
    }

           var stateEl = $('scheduleFolderState');
           var unsupportedEl = $('scheduleFolderUnsupported');
           var pickBtn = $('scheduleFolderPickBtn');
           if (!stateEl || !pickBtn) return;

             pickBtn.setAttribute('hidden', '');
             stateEl.setAttribute('hidden', '');
             if (unsupportedEl) unsupportedEl.classList.remove('is-hidden');

           pickBtn.removeAttribute('hidden');
           var name = readFolderName();

             stateEl.removeAttribute('data-i18n');
             stateEl.textContent = App.t('schedule.folder.repick').replace('{name}', name);
           } else {
             stateEl.setAttribute('data-i18n', 'schedule.folder.empty');
             stateEl.textContent = App.t('schedule.folder.empty');
           }

         async function applyFrequencyChange(newMode) {
           var ackedErr = $('schedulePasswordError');
           if (newMode !== 'off' && !readPasswordAcked()) {
             // D-18 enforcement
             if (ackedErr) ackedErr.classList.remove('is-hidden');
             // Revert the select to its previous value
             var sel = $('scheduleFrequencySelect');
             if (sel) sel.value = readScheduleMode();
             return false;
           }
           if (ackedErr) ackedErr.classList.add('is-hidden');

           // ON → OFF requires a confirm (UI-SPEC destructive confirm — neutral tone, not danger)
           var prev = readScheduleMode();
           if (prev !== 'off' && newMode === 'off') {
             var confirmed = await App.confirmDialog({
               titleKey: 'schedule.disableConfirm.title',
               messageKey: 'schedule.disableConfirm.body',
               confirmKey: 'schedule.disableConfirm.yes',
               cancelKey: 'schedule.disableConfirm.cancel',
               tone: 'neutral'
             });
             if (!confirmed) {
               var sel2 = $('scheduleFrequencySelect');
               if (sel2) sel2.value = prev;
               return false;
             }
           }

           try { localStorage.setItem('portfolioBackupScheduleMode', newMode); } catch (_) {}
           if (newMode === 'custom') {
             try { localStorage.setItem('portfolioBackupScheduleCustomDays', String(readCustomDays())); } catch (_) {}
           }
           refreshFrequencyHelper();
           refreshCustomDaysVisibility();
           return true;
         }

         function bindBackupsTab() {
           var sel = $('scheduleFrequencySelect');
           if (!sel) return; // tab not on this page
           sel.value = readScheduleMode();
           refreshFrequencyHelper();
           refreshCustomDaysVisibility();
           refreshFolderState();
           var customDays = $('scheduleCustomDays');
           if (customDays) customDays.value = String(readCustomDays());

           sel.addEventListener('change', function () { applyFrequencyChange(sel.value); });

           if (customDays) {
             customDays.addEventListener('change', function () {
               var n = Math.max(1, Math.min(365, Number(customDays.value) || 7));
               customDays.value = String(n);
               try { localStorage.setItem('portfolioBackupScheduleCustomDays', String(n)); } catch (_) {}
             });
           }

           var ack = $('schedulePasswordAcked');
           if (ack) {
             ack.checked = readPasswordAcked();

               // If unchecked AND schedule is non-off, force schedule back to off
               if (!ack.checked && readScheduleMode() !== 'off') {
                 sel.value = 'off';
                 applyFrequencyChange('off');
               }
               var err = $('schedulePasswordError');
               if (err) err.classList.add('is-hidden');
             });

           var pickBtn = $('scheduleFolderPickBtn');
           if (pickBtn) {

                 var handle = await BackupManager.pickBackupFolder();
                 if (handle && handle.name) {
                   try { localStorage.setItem('portfolioBackupFolderName', handle.name); } catch (_) {}
                   refreshFolderState();
                   App.showToast('', 'toast.autoBackupSet');
                 }
               } catch (err) {
                 console.error('Folder picker failed:', err);
               }
             });
           }
         }

         document.addEventListener('DOMContentLoaded', bindBackupsTab);
