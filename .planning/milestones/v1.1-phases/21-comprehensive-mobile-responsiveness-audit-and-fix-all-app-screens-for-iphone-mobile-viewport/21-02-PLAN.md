**Part B — Accordion JS toggle in add-session.js (per D-07):**

Add at the end of the DOMContentLoaded handler in add-session.js:

```javascript
// Accordion toggle — mobile only
const isMobileAccordion = window.matchMedia("(max-width: 768px)");

function initAccordions() {
  const sections = document.querySelectorAll(".accordion-section");
  if (!isMobileAccordion.matches) {
    // Desktop: ensure all sections are visible
    sections.forEach(s => s.classList.add("is-active"));
    return;
  }
  // Mobile: first section active by default
  sections.forEach((section, i) => {
    if (i === 0) section.classList.add("is-active");
    else section.classList.remove("is-active");
  });
}

document.querySelectorAll(".accordion-header").forEach(header => {
  header.addEventListener("click", () => {
    if (!isMobileAccordion.matches) return;
    const section = header.closest(".accordion-section");
    const wasActive = section.classList.contains("is-active");
    // Close all others
    document.querySelectorAll(".accordion-section").forEach(s => s.classList.remove("is-active"));
    // Toggle clicked
    if (!wasActive) section.classList.add("is-active");
  });
});

isMobileAccordion.addEventListener("change", initAccordions);
initAccordions();
```

**Part C — Mobile native date input swap (per D-06):**

In `assets/app.js`, find the `App.initBirthDatePicker` function. After it creates the 3-dropdown picker, add mobile detection:

```javascript
// Mobile: show native date input instead of dropdown picker
const mobileDate = window.matchMedia("(max-width: 768px)");
if (mobileDate.matches) {
  // Hide dropdown container, show native input
  const dropdownContainer = document.getElementById("birthDateContainer");
  if (dropdownContainer) {
    const nativeDateInput = document.createElement("input");
    nativeDateInput.type = "date";
    nativeDateInput.className = "input";
    nativeDateInput.id = "birthDateNative";
    // Read existing value from hidden input
    const hiddenInput = dropdownContainer.querySelector("input[type='hidden']") || document.getElementById("clientBirthDate") || document.getElementById("editClientBirthDate");
    if (hiddenInput && hiddenInput.value) {
      nativeDateInput.value = hiddenInput.value;
    }
    nativeDateInput.addEventListener("change", () => {
      if (hiddenInput) hiddenInput.value = nativeDateInput.value;
    });
    // Hide dropdowns, show native
    const selects = dropdownContainer.querySelectorAll("select");
    selects.forEach(s => s.style.display = "none");
    dropdownContainer.appendChild(nativeDateInput);
  }
}
```
