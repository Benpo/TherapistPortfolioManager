1. In assets/app.js, add these two shared functions inside the IIFE (before the return statement that exposes the public API), then expose them on the returned object:

   **formatSessionType(type):**
   ```javascript
   function formatSessionType(type) {
     const key = 'session.type.' + (type || 'clinic');
     return t(key);
   }
   ```
   This uses only the 3 valid session types: `clinic`, `online`, `other`. Do NOT include the old dead types `inPerson`, `proxy`, `surrogate`. The i18n keys `session.type.clinic`, `session.type.online`, `session.type.other` already exist in all language files.

   **readFileAsDataURL(file):**
   ```javascript
   function readFileAsDataURL(file) {
     return new Promise(function(resolve, reject) {
       const reader = new FileReader();
       reader.onload = function() { resolve(reader.result); };
       reader.onerror = function() { reject(reader.error); };
       reader.readAsDataURL(file);
     });
   }
   ```

   Add both to the returned public API object:
   ```javascript
   return { t, applyTranslations, setLanguage, getLanguage, formatSessionType, readFileAsDataURL };
   ```
