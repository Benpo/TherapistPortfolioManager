<objective>
Add the "Test backup password" dry-run safety net to the Backup & Restore modal. The user can drop an encrypted backup file + enter the password to verify the password decrypts the file — WITHOUT touching their current data.

Purpose: Closes the worst-case scenario where a therapist has a backup file, has forgotten the password, and only discovers this when they actually need to restore (mitigates Pitfall 2 from RESEARCH.md). Per D-12 the dry-run lives inside the Backup & Restore modal (RESEARCH.md Open Question 3 recommendation) under the Import section, where users are already in "I'm thinking about restoring" headspace.

Output: A new BackupManager.testBackupPassword(file, passphrase) public function that decrypts the .sgbackup to memory, validates the ZIP+manifest header, and returns a result object with NO IndexedDB writes and NO localStorage timestamp updates. New markup for the sub-card inside the modal (which Plan 02 left empty). 10 new i18n keys (backup.action.testPassword + 9 backup.testPassword.*) in 4 locales. Three regression tests with a reusable spy-instrumented PortfolioDB mock.

    async function testBackupPassword(file, passphrase) {
      if (!file) throw new Error(_t('backup.testPassword.invalid'));
      const ext = (file.name || '').split('.').pop().toLowerCase();
      if (ext !== 'sgbackup') {
        throw new Error(_t('backup.testPassword.notEncrypted'));
      }
      let zipBlob;
      try {
        zipBlob = await _decryptBlob(file, passphrase);
      } catch (err) {
        if (err && err.name === 'OperationError') {
          throw new Error(_t('backup.testPassword.wrongPassphrase'));
        }
        throw err;
      }
      if (!zipBlob) {
        throw new Error(_t('backup.testPassword.invalid'));
      }
      const arrayBuffer = await zipBlob.arrayBuffer();
      const zip = await JSZip.loadAsync(arrayBuffer);
      const jsonFile = zip.file('backup.json');
      if (!jsonFile) {
        throw new Error(_t('backup.testPassword.invalid'));
      }
      const text = await jsonFile.async('string');
      let manifest;
      try {
        manifest = JSON.parse(text);
      } catch (_) {
        throw new Error(_t('backup.testPassword.invalid'));
      }
      // No PortfolioDB writes, no localStorage writes. Return memory-only result.
      return {
        ok: true,
        manifestVersion: manifest.version || null,
        exportedAt: manifest.exportedAt || null,
        clientCount: Array.isArray(manifest.clients) ? manifest.clients.length : 0,
        sessionCount: Array.isArray(manifest.sessions) ? manifest.sessions.length : 0
      };
    }
