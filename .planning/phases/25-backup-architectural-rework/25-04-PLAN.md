Chip state derivation (pure function — for unit testability):
  getChipState({ now, lastExport, intervalMs }) returns one of:
    'never'   when lastExport is null/undefined/NaN
    'fresh'   when (now - lastExport) <= threshold_fresh
    'warning' when threshold_fresh < (now - lastExport) <= threshold_danger
    'danger'  when (now - lastExport) > threshold_danger

  Thresholds:
    schedule OFF (intervalMs is null): threshold_fresh = 7 days; threshold_danger = 14 days
    schedule ON (intervalMs > 0):       threshold_fresh = intervalMs; threshold_danger = intervalMs * 2
    The "warning" zone is (threshold_fresh, threshold_danger]; UI-SPEC says interval×1.5 demarcates warning→danger BUT the simpler rule "danger > 2×" matches CONTEXT D-14 ("danger > interval × 2"); use the CONTEXT rule. The intermediate "interval×1.5" is the lower bound of the warning zone in UI-SPEC (warning ≤ interval × 1.5 means warning extends to interval×1.5, danger above). Reconcile by using:
      threshold_warning_max = intervalMs * 1.5  (warning ends here)
      threshold_danger      = intervalMs * 2    (danger starts here)
    State logic:
      elapsed <= intervalMs → 'fresh'
      elapsed <= intervalMs * 1.5 → 'warning'
      elapsed > intervalMs * 1.5 → 'danger' until elapsed > intervalMs * 2 (still 'danger' — there is no separate "very-danger")
    For the test table, treat the boundary as: danger when elapsed > 1.5 * intervalMs (NOT > 2×). This matches UI-SPEC's clearer phrasing "warning ≤ interval × 1.5 / danger > interval × 2" by treating the intermediate (1.5,2] zone as still danger (more danger). Document the simplification in the SUMMARY: "warning is (interval, 1.5×interval]; everything beyond 1.5×interval is danger" — this collapses CONTEXT D-14's three thresholds to two while preserving the intent (positive reinforcement + escalating concern).

    var SCHEDULE_INTERVAL_MS = {
      daily:   24 * 60 * 60 * 1000,
      weekly:  7 * 24 * 60 * 60 * 1000,
      monthly: 30 * 24 * 60 * 60 * 1000
    };

    function getScheduleIntervalMs() {
      var mode;
      try { mode = localStorage.getItem('portfolioBackupScheduleMode') || 'off'; }
      catch (_) { return null; }
      if (mode === 'off') return null;
      if (mode === 'custom') {
        var raw = Number(localStorage.getItem('portfolioBackupScheduleCustomDays'));
        var days = (raw && raw > 0) ? raw : 7;
        return days * 24 * 60 * 60 * 1000;
      }
      return SCHEDULE_INTERVAL_MS[mode] || null;
    }

    function getChipState(opts) {
      opts = opts || {};
      var now = (typeof opts.now === 'number') ? opts.now : Date.now();
      var lastExport = opts.lastExport;
      if (lastExport === null || typeof lastExport === 'undefined' || isNaN(lastExport)) {
        return 'never';
      }
      var intervalMs = (typeof opts.intervalMs === 'number') ? opts.intervalMs : null;
      var elapsed = now - Number(lastExport);
      var DAY = 24 * 60 * 60 * 1000;
      var freshLimit, warningLimit;
      if (intervalMs === null) {
        freshLimit = 7 * DAY;
        warningLimit = 14 * DAY;
      } else {
        freshLimit = intervalMs;
        warningLimit = intervalMs * 1.5;
      }
      if (elapsed <= freshLimit) return 'fresh';
      if (elapsed <= warningLimit) return 'warning';
      return 'danger';
    }
