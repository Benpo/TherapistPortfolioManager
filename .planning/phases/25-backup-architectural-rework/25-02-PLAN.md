       function formatRelativeTime(timestampMs) {
         if (!timestampMs || isNaN(timestampMs)) return null;
         const elapsed = Date.now() - timestampMs;
         const rtf = new Intl.RelativeTimeFormat(localStorage.getItem('portfolioLang') || 'en', { numeric: 'auto' });
         const dayMs = 24*60*60*1000;
         const hourMs = 60*60*1000;
         const minMs = 60*1000;
         if (elapsed >= dayMs) return rtf.format(-Math.floor(elapsed / dayMs), 'day');
