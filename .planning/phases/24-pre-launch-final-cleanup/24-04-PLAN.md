       async function seedSnippetsIfNeeded(db) {
         if (_seedingDone) return;
         _seedingDone = true;
         if (!window.SNIPPETS_SEED || !Array.isArray(window.SNIPPETS_SEED)) {
           console.warn("snippets: SNIPPETS_SEED not loaded; skipping seed");
           return;
         }
         const deletedSeedIds = await readDeletedSeedIds(db);
         const existingSeedIds = new Set();
         await new Promise((resolve, reject) => {
           const tx = db.transaction("snippets", "readonly");
           const store = tx.objectStore("snippets");
           const idx = store.index("origin");
           const req = idx.openCursor(IDBKeyRange.only("seed"));
           req.onsuccess = (e) => {
             const cursor = e.target.result;
             if (cursor) { existingSeedIds.add(cursor.value.id); cursor.continue(); }
             else resolve();
           };
           req.onerror = () => reject(req.error);
         });
         const toInsert = window.SNIPPETS_SEED.filter(s => !deletedSeedIds.has(s.id) && !existingSeedIds.has(s.id));
         if (toInsert.length === 0) return;
         await new Promise((res, rej) => {
           const tx = db.transaction("snippets", "readwrite");
           const store = tx.objectStore("snippets");
           for (const s of toInsert) {
             store.add({ ...s, trigger: s.trigger.toLowerCase() });
           }
           tx.oncomplete = res;
           tx.onerror = () => rej(tx.error);
         });
       }

       async function readDeletedSeedIds(db) {
         try {

           return new Set();
         }
