     ```javascript
     // After the before and after severity scales are rendered for an issue:
     function updateDelta(issueBlock) {
       const beforeValue = App.getSeverityValue(issueBlock.querySelector('[id*="Before"]') || issueBlock.querySelector('.severity-before'));
       const afterValue = App.getSeverityValue(issueBlock.querySelector('[id*="After"]') || issueBlock.querySelector('.severity-after'));
       let deltaEl = issueBlock.querySelector('.severity-delta');
       if (!deltaEl) {
         deltaEl = document.createElement('span');
         deltaEl.className = 'severity-delta';
         // Append after the after-severity row
         issueBlock.querySelector('.issue-severity-row:last-of-type')?.appendChild(deltaEl);
       }
       if (beforeValue !== null && afterValue !== null && beforeValue !== undefined && afterValue !== undefined) {
         const delta = afterValue - beforeValue;
         const sign = delta > 0 ? '+' : '';
         deltaEl.textContent = sign + delta;
         deltaEl.classList.toggle('delta-positive', delta > 0);
         deltaEl.classList.toggle('delta-negative', delta < 0);
         deltaEl.style.display = delta === 0 ? 'none' : '';
       } else {
         deltaEl.style.display = 'none';
       }
     }
     ```
